üìê Responsive Design

1. Breakpoints

$breakpoints: (
  "mobile": 320px,
  "tablet": 768px,
  "desktop": 1024px,
  "wide": 1440px,
);

@mixin respond-to($breakpoint) {
  @if map-has-key($breakpoints, $breakpoint) {
    @media (min-width: map-get($breakpoints, $breakpoint)) {
      @content;
    }
  }
}

const TradingPanel: React.FC = () => {
  const handleKeyPress = (e: KeyboardEvent) => {
    switch (e.key) {
      case "ArrowUp":
        incrementPrice();

      case "ArrowDown":
        decrementPrice();
        break;
      case "Enter":
        submitOrder();
